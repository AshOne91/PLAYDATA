# 함수에 대한 정의 부분
```
def add(a, b):
  return a + b;

def substract(a, b):
  return a - b;

def multiply(a, b):
  return a * b;

def divide(a, b):
  return a / b;

add_result = add(5, 3)
print(add_result)

multiply_result = multiply(a = 5, b = 3)
divide(5, b=3)

help(add)
help(len)
print(add.__doc__)

```
